\begin{HWProblem}[5]
Prove that $(a + b)^2 = a^2 + 2ab + b^2$.
\end{HWProblem}

\begin{HWProof}
$(a + b)^2 = (a + b) \cdot (a + b) = a^2 + ba + ab + b^2 = a^2 + 2ab + b^2$
\end{HWProof}
